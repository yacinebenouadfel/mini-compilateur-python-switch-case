# Fonction avec switch
def evaluer_note(note):
    if note < 0 or note > 20:
        return "Note invalide"

    switch (note) {
        case 20:
            return "Excellent"
        case 18:
        case 19:
            return "Très bien"
        case 16:
        case 17:
            return "Bien"
        case 14:
        case 15:
            return "Assez bien"
        default:
            return "Passable"
    }
